# 421. Maximum XOR of Two Numbers in an Array
`n` - number of elements in the input array

`m` - min(bits required to store max(input array))

## 1.Python Trie (2 pass)

Space - `O(1)` since `O(2^m) = constant` 

Time - `O(n)` 

```py
class Solution:
    def findMaximumXOR(self, nums: List[int]) -> int:
        trieSt = dict()
        maxLength = len(bin(max(nums))[2:])
        
        numsAl = [
            [(num>>i)&1 for i in range(maxLength)[::-1]] 
            for num in nums
            ]
            
        maxXor = 0
        for num in numsAl:
            current = trieSt
            for bit in num:
                if not (bit in current):
                    current[bit] = dict()
                current = current[bit]
        
        
        for num in numsAl:
            currentXr = 0
            current = trieSt
            for bit in num:
                currentXr <<= 1
                bitOpp = 1-bit
                if bitOpp in current:
                    currentXr += 1
                    current = current[bitOpp]
                else:
                    current = current[bit]
            maxXor = max(maxXor,currentXr)
        
        return maxXor
```

## 2. Python Trie (1 pass)

```py
class Solution:
    
    @staticmethod
    def getBinNums(nums: List[List[int]], maxPad) -> List[int]:
        resultNums = []
        for num in nums:
            tempNum = []
            for i in range(maxPad)[::-1]:
                tempNum.append((num>>i)&1)
            resultNums.append(tempNum)
        return resultNums

    def findMaximumXOR(self, nums: List[int]) -> int:
        maxPad = len(bin(max(nums))[2:])        
        binNumsList = self.getBinNums(nums, maxPad)
        trieStruct = dict()
        maxXor = 0
        
        for num in binNumsList:
            current = trieStruct
            currXor = 0
            xorPos = trieStruct
            for bit in num:                
                oppBit = 1-bit
                currXor= currXor<<1
                
                #insert
                if not (bit in current):
                    current[bit] = dict()
                current = current[bit]
                
                # search opposite
                if oppBit in xorPos:
                    currXor |= 1
                    xorPos = xorPos[oppBit]
                else:
                    xorPos = xorPos[bit]
                
            maxXor = max(maxXor, currXor)
        return maxXor
```
